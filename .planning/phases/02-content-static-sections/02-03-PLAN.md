```typescript
export const packageSchema = z.object({
  id: z.enum(["small", "medium", "large"]),
  name: z.string(),
  guestRange: z.object({ min, max }).refine(r => r.min <= r.max),
  pricePerPerson: z.object({ min, max }).refine(r => r.min <= r.max),
  includes: z.array(z.string()).min(1),

  order: z.number().int(),
});
```
