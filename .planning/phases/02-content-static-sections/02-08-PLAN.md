### `src/components/GalleryGrid.tsx`

```tsx
import { Suspense, lazy, useState } from "react";
import { MasonryPhotoAlbum } from "react-photo-album";
import "react-photo-album/masonry.css";
import { cn } from "@/lib/utils";

// Dynamic imports — lightbox ships only when user opens it (GAL-06 + PATTERNS anti-pattern #15)
const Lightbox = lazy(() => import("yet-another-react-lightbox"));
const Captions = lazy(() =>
  import("yet-another-react-lightbox/plugins/captions").then((m) => ({ default: m.default })),
);
const Counter = lazy(() =>
  import("yet-another-react-lightbox/plugins/counter").then((m) => ({ default: m.default })),
);

const INITIAL = 10;

type AspectRatio = "1:1" | "4:3" | "3:2" | "16:9" | "3:4" | "2:3" | "9:16";

interface GalleryPhoto {
  image: string;
  alt: string;
  caption?: string;
  aspectRatio: AspectRatio;
  order: number;
}

interface Props {
  photos: GalleryPhoto[];
}

const ratioToDimensions = (r: AspectRatio): { width: number; height: number } => {
  const [w, h] = r.split(":").map(Number) as [number, number];
  const base = 1200;
  return { width: base, height: Math.round((base * h) / w) };
};

export default function GalleryGrid({ photos }: Props) {
  const [showAll, setShowAll] = useState(false);
  const [lightboxIndex, setLightboxIndex] = useState<number | null>(null);

  const albumPhotos = photos.map((p) => {
    const { width, height } = ratioToDimensions(p.aspectRatio);
    return {
      src: p.image,
      width,
      height,
      alt: p.alt,
      ...(p.caption ? { title: p.caption } : {}),
    };
  });

  const visible = showAll ? albumPhotos : albumPhotos.slice(0, INITIAL);

  const lightboxSlides = photos.map((p) => ({
    src: p.image,
    alt: p.alt,
    ...(p.caption ? { description: p.caption } : {}),
  }));

  return (
    <>
      <MasonryPhotoAlbum
        photos={visible}
        columns={(containerWidth) =>
          containerWidth < 640 ? 2 : containerWidth < 1024 ? 3 : 4
        }
        spacing={16}
        onClick={({ index }) => setLightboxIndex(index)}
      />

      {albumPhotos.length > INITIAL && (
        <div className="flex justify-center mt-12">
          <button
            type="button"
            onClick={() => setShowAll((v) => !v)}
            className={cn(
              "inline-flex items-center justify-center rounded-full border border-ink/20 text-ink",
              "px-6 py-3 text-body-md font-semibold",
              "hover:bg-ink hover:text-surface transition-colors duration-200 motion-reduce:transition-none",
              "focus-visible:outline-2 focus-visible:outline-primary focus-visible:outline-offset-4",
            )}
          >
            {showAll
              ? "Show less"
              : `View the full gallery (${albumPhotos.length})`}
          </button>
        </div>
      )}

      {lightboxIndex !== null && (
        <Suspense fallback={null}>
          <Lightbox
            open={lightboxIndex !== null}
            close={() => setLightboxIndex(null)}
            slides={lightboxSlides}
            index={lightboxIndex}
            plugins={[Captions as any, Counter as any]}
            animation={{ fade: 0 }}
            controller={{ closeOnBackdropClick: true }}
          />
        </Suspense>
      )}
    </>
  );
}
```

Critical contracts:
- Hydration: `client:visible` on GallerySection's GalleryGrid mount (GAL-06 + UI-SPEC § Island Hydration).
- Lightbox dynamically imported via `lazy()` — ships only when `lightboxIndex !== null` triggers Suspense mount (PATTERNS.md anti-pattern #15 + #3).
- INITIAL reveal locked at 10 per UI-SPEC § 6.
- `react-photo-album` uses `width` and `height` derived from `aspectRatio` enum — actual pixel dimensions are irrelevant; the RATIO is what matters for CLS prevention (GAL-02).
- Columns responsive: 2 (mobile) / 3 (tablet) / 4 (desktop).
- Button label toggles: `View the full gallery (N)` ↔ `Show less` (UI-SPEC § Copywriting locked).
- Lightbox animation fade = 0 (instant) — UI-SPEC § Interaction Patterns tells us to honor reduced-motion; rather than detecting the media query at runtime, set it to 0 always (acceptable per UI-SPEC; avoids a `matchMedia` call). Users with motion preferences lose 300ms fade but gain consistency. Alternative: `animation={{ fade: prefersReducedMotion ? 0 : 300 }}` if media query is imported from a helper; simpler pattern above is acceptable for Phase 2.
- Captions + Counter plugins enabled (UI-SPEC § 6).
- Thumbnail click opens lightbox at matching index (GAL-04 accessible lightbox).
- Keyboard (← → Escape) handled by library default.
- Focus return to trigger on close: library default behavior (Suspense unmount returns focus to the last-focused element; verify in dev but library docs confirm this is default).
