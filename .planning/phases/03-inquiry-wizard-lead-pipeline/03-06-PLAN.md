```typescript
// Source: LEAD-11 + RESEARCH §Pattern 10 + §Open Question Q3 accepted fallback + PATTERNS §retry-email.ts
export const prerender = false;

import type { APIRoute } from "astro";
import { getLeadStore } from "@/lib/leads/store";
import { sendLeadNotification, sendLeadConfirmation } from "@/lib/email/send";

const MAX_RETRIES = 3;
const MIN_AGE_MS = 60 * 60 * 1000; // 1 hour — don't retry recent failures that might recover on their own

export const GET: APIRoute = async ({ request }) => {
  const auth = request.headers.get("authorization");
  const expected = `Bearer ${import.meta.env.CRON_SECRET}`;
  if (!auth || auth !== expected) {
    return new Response(JSON.stringify({ error: "unauthorized" }), { status: 401, headers: { "content-type": "application/json" } });
  }

  const store = getLeadStore();
  const pending = await store.findPendingEmails({ maxRetries: MAX_RETRIES, minAgeMs: MIN_AGE_MS });
  let sent = 0;
  let failed = 0;

  for (const lead of pending) {
    if (lead.notifyEmailStatus === "pending") {
      try { await sendLeadNotification(lead); await store.markEmailRetry(lead.submissionId, "notify", "sent"); sent++; }
      catch (err) { await store.markEmailRetry(lead.submissionId, "notify", "failed"); failed++; console.error("retry_notify_failed", { submissionId: lead.submissionId, reason: String(err) }); }
    }
    if (lead.confirmEmailStatus === "pending") {
      try { await sendLeadConfirmation(lead); await store.markEmailRetry(lead.submissionId, "confirm", "sent"); sent++; }
      catch (err) { await store.markEmailRetry(lead.submissionId, "confirm", "failed"); failed++; console.error("retry_confirm_failed", { submissionId: lead.submissionId, reason: String(err) }); }
    }
  }

  return new Response(JSON.stringify({ scanned: pending.length, sent, failed }), { status: 200, headers: { "content-type": "application/json" } });
};
```

```typescript
// Source: LEAD-12 + RESEARCH §Pitfall 8 (HMAC timingSafeEqual) + §Pattern §LEAD-12
// Correlation contract: reads `event.data.tags.submission_id` + `event.data.tags.which` — emitted by Plan 05's send.ts.
// If Plan 05's tag emission regresses (keys renamed), this handler will log `resend_webhook_missing_tags` and no-op.
export const prerender = false;

import type { APIRoute } from "astro";
import { createHmac, timingSafeEqual } from "node:crypto";
import { getLeadStore } from "@/lib/leads/store";

interface ResendEvent {
  type: "email.delivered" | "email.bounced" | "email.complained" | "email.delivery_delayed" | "email.sent" | "email.opened" | "email.clicked" | string;
  data: { email_id?: string; to?: string[]; subject?: string; tags?: Record<string, string> };
  created_at?: string;
}

function verifySignature(rawBody: string, signatureHeader: string | null): boolean {
  if (!signatureHeader) return false;
  const secret = import.meta.env.RESEND_WEBHOOK_SECRET;
  if (!secret) return false;
  const expected = createHmac("sha256", secret).update(rawBody).digest("hex");
  // Header format per Resend docs may be `sha256=<hex>`; accept both bare hex and prefixed forms.
  const provided = signatureHeader.startsWith("sha256=") ? signatureHeader.slice(7) : signatureHeader;
  const a = Buffer.from(expected, "hex");
  const b = Buffer.from(provided, "hex");
  if (a.length !== b.length) return false;
  return timingSafeEqual(a, b);
}

export const POST: APIRoute = async ({ request }) => {
  const raw = await request.text();
  const signatureHeader = request.headers.get("x-resend-signature") ?? request.headers.get("resend-signature");

  if (!verifySignature(raw, signatureHeader)) {
    return new Response(JSON.stringify({ error: "invalid_signature" }), { status: 401, headers: { "content-type": "application/json" } });
  }

  let event: ResendEvent;
  try { event = JSON.parse(raw); }
  catch { return new Response(JSON.stringify({ error: "invalid_json" }), { status: 400, headers: { "content-type": "application/json" } }); }

  // Correlation tags emitted by Plan 05 send.ts: { submission_id, which: "notify" | "confirm" }.
  const submissionId = event.data?.tags?.["submission_id"];
  const which = event.data?.tags?.["which"] as "notify" | "confirm" | undefined;
  if (!submissionId || !which) {
    // Still return 200 so Resend stops retrying; log for investigation.
    console.warn("resend_webhook_missing_tags", { type: event.type });
    return new Response(JSON.stringify({ ok: true, handled: false }), { status: 200, headers: { "content-type": "application/json" } });
  }

  const store = getLeadStore();
  const status =
    event.type === "email.delivered" || event.type === "email.sent" ? "sent"
    : event.type === "email.bounced" || event.type === "email.complained" || event.type === "email.delivery_delayed" ? "failed"
    : null;
  if (status) {
    try { await store.markEmailRetry(submissionId, which, status); }
    catch (err) { console.error("resend_webhook_update_failed", { submissionId, reason: String(err) }); }
  }

  return new Response(JSON.stringify({ ok: true }), { status: 200, headers: { "content-type": "application/json" } });
};
```

**CRITICAL — B5 Turnstile test-key isolation strategy (SPAM-06 contradiction resolution):**

Plan 02's `scripts/check-turnstile-keys.sh dist` blocks the Cloudflare-published Turnstile test keys (`1x00000000000000000000AA`, `1x0000000000000000000000000000000AA`, etc.) from appearing in the production `dist/` artifact. Astro inlines `PUBLIC_*` env vars at build time. If these test keys are present in a `.env` or `.env.production` file during `pnpm build`, the site key (`PUBLIC_TURNSTILE_SITE_KEY`) gets baked into the client bundle and the SPAM-06 CI gate fails.

**Therefore Playwright MUST NOT use `.env` files to configure Turnstile.** Test keys are injected via one of two process-scoped paths:

1. **Preferred — `playwright.config.ts` `webServer.env`:** the `webServer.env` object is passed as OS environment variables to the subprocess that runs `pnpm build && pnpm preview`. Astro reads `import.meta.env.PUBLIC_TURNSTILE_SITE_KEY` via Vite's env loader, which prioritizes process env over `.env` files. The resulting `dist/` contains the test site key because Astro inlines it — BUT this `dist/` exists only in the Playwright subprocess directory and is NEVER committed or deployed. Production deploys run `pnpm build` without these env vars set (Vercel env UI is the source of truth), and the SPAM-06 gate is clean.
   - Caveat: the Playwright subprocess `dist/` WILL contain the test keys (that's the point — the test site exercises the widget). If the local developer runs `check-turnstile-keys.sh dist` after a Playwright run, it will flag. The SPAM-06 script is intended for production deploys (Vercel build output), not local E2E scratch builds. Document this explicitly in `scripts/check-turnstile-keys.sh` usage notes (OUT-OF-SCOPE for this plan — Plan 02 owns the script).

2. **Alternative — `PLAYWRIGHT_BASE_URL` against Vercel Preview:** point Playwright at a deployed Vercel Preview URL whose test keys are set via the Vercel Preview env UI (not the repo). Vercel Preview's `dist/` has the test keys (expected — preview is disposable); production's `dist/` (built separately for main branch) has real keys. SPAM-06 CI gate runs only against production builds, so it stays green.

**Choose Strategy 1 for this task's default (local CI).** Strategy 2 is an override via the `PLAYWRIGHT_BASE_URL` env var.
